		def __init__(self, name):
			self.name = name
		def speak(self):
			print("I am an animal and my name is " + self.name + ".") 
		def eat(self):
			print("I am eating.")

		def __init__(self, name, breed):
			super().__init__(name)

			print("Woof woof!")

		dog.speak()
		dog.eat()
		dog.bark()
